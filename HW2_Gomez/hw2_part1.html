function plot_avgs(x)
avg_vec = [1 1 1]./3;
conv_sig = conv(x,avg_vec);
plot(1:1:length(x),x,'b',2:1:(length(conv_sig)-1),conv_sig(2:end-1),'r');
title('combined signal and moving avg')
xlabel('samples n')
ylabel('value')
grid on
legend('original','avg')
end
 
 
% difference eqs
function y = diffeq_1(x)
y = zeros(size(x));
y(1) = 0;
for n = 2:length(x)
y(n) = .8*(y(n-1)) + x(n);
end
